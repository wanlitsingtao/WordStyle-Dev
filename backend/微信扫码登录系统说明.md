### 2. 首次登录逻辑

```python
# 检查用户是否存在
user = db.query(User).filter(User.wechat_openid == openid).first()

if not user:
    # 首次登录，创建新用户
    free_paragraphs = get_free_paragraphs_config(db)  # 从配置读取
    
    user = User(
        wechat_openid=openid,
        wechat_nickname=nickname,
        wechat_avatar=avatar,
        paragraphs_remaining=free_paragraphs,  # 赠送免费额度
        balance=0.0,
        is_active=True
    )
    db.add(user)
    db.commit()
    
    message = f"欢迎！已赠送您 {free_paragraphs} 段免费额度"
else:
    # 老用户，更新信息
    user.wechat_nickname = nickname
    user.wechat_avatar = avatar
    db.commit()
    
    message = f"欢迎回来，{user.wechat_nickname}"
```
